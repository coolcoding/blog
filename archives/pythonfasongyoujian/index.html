def send_mail(mail_list,sub,content): msg=email.MIMEText.MIMEText(content,_subtype=’plain’,_charset=’gb2312’) msg[‘Subject’] = sub msg[‘From’] = ‘ops@dongwm.com’ msg[‘To’] = “;”.join(mail_list) msg[‘date’]=time.ctime() me = mail_user+’<’+mail_user+’@’+’dongwm.com+’>’ body=email.MIMEText.MIMEText(content,_charset=’gb2312’) bb =msg.as_string()+body.as_string() s = smtplib.SMTP(mail_server, mail_server_port) s.sendmail(me,mail_list,msg.as_string()) s.quit()
